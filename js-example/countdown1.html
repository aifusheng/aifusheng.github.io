使用JavaScript显示过年倒计时

要实现JavaScript显示过年倒计时,解决思路是:用结束的时间减去当前的时间

查看演示 下载源码

下面是js源代码：

/*当网页全部加载完后,开始执行showTime()函数*/
window.onload=function(){
showTime();
}
function checkTime(i){
if (i<10){
i='0'+i;
}
return i;
}
function showTime(){
	/*当前时间*/
	var currenttime=new Date();
	/*结束时间*/
	var endtime=new Date("2017,1,27;00:00:00");
	var lefttime=parseInt((endtime.getTime()-currenttime.getTime())/1000);
	var days=parseInt(lefttime/(24*60*60));
	var hours=parseInt(lefttime/(60*60)%24);
	var minutes=parseInt(lefttime/60%60);
	var seconds=parseInt(lefttime%60);
	days=checkTime(days);
	hours=checkTime(hours);
	minutes=checkTime(minutes);
	seconds=checkTime(seconds);
	document.getElementById('showtime').innerHTML="距离过年还有:"+days+"天"+hours+'小时'+minutes+'分钟'+seconds+'秒';
	/*判断是否过年了*/
	if (lefttime<=0){
document.getElementById('showtime').innerHTML="啦啦啦!!!---过年了!!!";
}
	setTimeout(showTime,500);
}
